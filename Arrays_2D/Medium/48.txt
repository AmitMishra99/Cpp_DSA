#include <iostream>
#include<algorithm>
using namespace std;
int main() {

    int m;
    cout<<"Enter a Number of Rows : ";
    cin>>m;
    int n;
    cout<<"Enter a Number of Columns : ";
    cin>>n;
    int mat[m][n]; 
    
    // Taking input for the first matrix
    cout << "Enter Elements of Matrix: "<<endl;
    for(int i=0;i<m;i++) {
        for(int j=0;j<n;j++) {
            cin>>mat[i][j];
        }
    }
    cout<<endl;

    // Into the Same Matrix 
    for(int i = 0; i < m; i++) {
        for(int j = i+1; j < n; j++) {
            if(i==j) continue;
            swap(mat[i][j] , mat[j][i]);
        }
    }

    for(int k=0;k<m;k++){
        int i=0;
        int j=m-1;
        while (i<=j){
                swap( mat[k][i] , mat[k][j]);
                i++;
                j--;
        }
    }

    cout << "Transpse of given Matrix is: "<<endl;
    for(int i=0;i<n;i++) {
        for(int j=0;j<m;j++) {
            cout<<mat[i][j]<<" ";
        }
        cout<<endl;
    }

}
