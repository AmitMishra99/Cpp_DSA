#include <iostream>
#include<algorithm>
using namespace std;
int main() {

    int m;
    cout<<"Enter a Number of Rows : ";
    cin>>m;
    int n;
    cout<<"Enter a Number of Columns : ";
    cin>>n;
    int arr[m][n]; 
    
    // Taking input for the first matrix
    cout << "Enter Elements of Matrix: "<<endl;
    for(int i=0;i<m;i++) {
        for(int j=0;j<n;j++) {
            cin>>arr[i][j];
        }
    }
    cout<<endl;

    int minr = 0;
    int maxr = m-1;
    int minc = 0;
    int maxc = n-1;

    while(minr<=maxr && minc<=maxc){
   // Left to right  
   if(minr>maxr || minc>maxc) break;
    for(int i=minc;i<=maxc;i++){
        cout<<arr[minr][i]<< ' ';
    }
    minr++;

   if(minr>maxr || minc>maxc) break;
    // Top to bottom  
    for(int i=minr;i<=maxr;i++){
        cout<<arr[i][maxc]<< ' ';
    }
    maxc--;

   if(minr>maxr || minc>maxc) break;
    // Right to left  
    for(int i=maxc;i>=minc;i--){
        cout<<arr[maxr][i]<< ' ';
    }
    maxr--;

   if(minr>maxr || minc>maxc) break;
    // Bottom to top  
    for(int i=maxr;i>=minr;i--){
        cout<<arr[i][minc]<< ' ';
    }
    minc++;

    }

//     Also if u want to add the condtion create

}
