#include<iostream>
#include<vector>
using namespace std;
int main(){


    vector<vector<int>>v1;
    
    for(int i=0;i<5;i++){
        vector<int>v;
        for(int j=0;j<=i;j++){
            if(j==0 || j==i)  v.push_back(1);
            else v.push_back(v1[i-1][j] + v1[i-1][j-1]);
        }
        v1.push_back(v);
    }

    int m = v1.size();

     for(int i=0;i<m;i++){
        for(int j=0;j<v1[i].size();j++){
            cout<<v1[i][j]<<" ";
        }
        cout<<endl;
    }

    
}
